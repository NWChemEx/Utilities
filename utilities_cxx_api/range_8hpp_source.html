104template<typename ElementType>
105Range<ElementType>::Range(ElementType range_begin, ElementType range_end,
106 size_type inc) noexcept :
107 m_start_(range_begin),
108 m_size_([=]() {
109 const auto lesser = std::min(range_begin, range_end);
110 const auto higher = std::max(range_begin, range_end);
111 const auto remainder = (higher - lesser) % inc;
112 return (higher - lesser - remainder) / inc + (remainder > 0);
113 }()),
114 m_inc_(range_begin <= range_end ? inc : -1 * inc) {
115 static_assert(std::is_integral_v<ElementType>,
116 "T for Range<T> must be an integral type.");
117}
