33template<typename T>
34std::ostream& print_element(std::ostream& os, T&& value) {
35 using clean_T = std::decay_t<T>;
36 if constexpr(type_traits::is_printable_v<clean_T>) {
37 return os << std::forward<T>(value);
38 } else {
39 return os << "<" << typeid(clean_T).name() << " " << &value << ">";
40 }
41}

43template<typename T, typename Fxn>
44std::ostream& print_queue(std::ostream& os, Fxn&& f, T&& rhs) {
45 os << '[';
46 bool printed = false;
47 while(!rhs.empty()) {
48 if(printed) os << ", ";
49 print_element(os, f(rhs));
50 rhs.pop();
51 printed = true;
52 }
53 return os << ']';
54}

56template<typename T>
57std::ostream& print_list(std::ostream& os, T&& rhs, char odelim, char cdelim) {
58 os << odelim;
59 bool printed = false;
60 for(const auto& x : rhs) {

68template<typename T>
69std::ostream& print_associative(std::ostream& os, T&& rhs, char odelim,
70 char cdelim) {
71 os << odelim;
72 bool printed = false;
73 for(const auto& x : rhs) {
74 if(printed) os << ", ";
75 os << "(";
76 print_element(os, x.first) << " : ";
77 print_element(os, x.second) << ")";
78 printed = true;
79 }
80 return os << cdelim;
81}

83template<std::size_t depth, typename T>
84std::ostream& print_tuple(std::ostream& os, T&& rhs, char odelim, char cdelim) {
85 constexpr auto max_depth = std::tuple_size_v<std::decay_t<T>>;
86 if constexpr(depth == max_depth && depth == 0) {
87 return os << odelim << cdelim;
88 } else if constexpr(depth == max_depth) {
89 return os << cdelim;
90 } else if constexpr(depth == 0) {
91 os << odelim;
92 print_element(os, std::get<0>(rhs));
93 return print_tuple<depth + 1>(os, std::forward<T>(rhs), odelim, cdelim);
94 } else {
95 os << ", ";
96 print_element(os, std::get<depth>(rhs));
97 return print_tuple<depth + 1>(os, std::forward<T>(rhs), odelim, cdelim);
98 }
99}
