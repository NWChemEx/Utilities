273template<typename DerivedType>
274bool operator==(const IndexableContainerBase<DerivedType>& lhs,
275 const IndexableContainerBase<DerivedType>& rhs) noexcept {
276 if(lhs.size() != rhs.size()) return false;
277 /* Note that here we don't use std::equal because it assumes that the return
278 * value of dereferencing an iterator is a true reference. Consequentially,
279 * we'll get segfaults if the user's container returns things by temporary
280 * copies.
281 */
282 for(decltype(lhs.size()) counter = 0; counter < lhs.size(); ++counter)
283 if(lhs[counter] != rhs[counter]) return false;
284 return true;
285}

344 msg += std::to_string(size()) + ").";
