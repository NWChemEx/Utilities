210inline std::ostream& operator<<(std::ostream& os, const Timer& t) {
211 for(const auto& [desc, dt] : t) {
212 using namespace std::chrono;
213 auto h = duration_cast<hours>(dt);
214 auto m = duration_cast<minutes>(dt) - h;
215 auto s = duration_cast<seconds>(dt) - h - m;
216 auto ms = duration_cast<milliseconds>(dt) - h - m - s;
217 os << desc << " : " << h.count() << " h " << m.count() << " m "
218 << s.count() << " s " << ms.count() << " ms" << std::endl;
219 }
220 return os;
221}
