62  // This function does the heavy lifting
63  std::streamsize xsputn(const char* s, std::streamsize n) override {
64  // Break paragraph into sentences
65  auto sentences = strings::split_string(std::string(s, s + n), "\n");
66 
67  const auto nsentences = sentences.size();
68  for(std::size_t si = 0; si < nsentences; ++si) { // Loop over sentences
69  auto sentence = sentences[si];
70  while(!sentence.empty()) {
71  const unsigned short size = sentence.size();
72  if(m_nchars_ + size <= m_w_) { // Whole thing fits
73  (*m_os_) << sentence;
74  m_nchars_ += size;
75  break;
76  }
77 
78  // Remainder of this while figures out where to break the line
79 
80  // How many characters do we have left?
81  const auto char_left = m_w_ - m_nchars_;
82 
83  // Find the first space we can break on
84  std::size_t offset = sentence.find(' ', 0);
85 
86  // Was there any spaces left?
87  const bool no_spaces_left = (offset == std::string::npos);
88 
89  // Aren't on empty line and can't print any part of what remains
90  if((no_spaces_left || offset > char_left) && m_nchars_ > 0) {
91  (*m_os_) << std::endl;
92  m_nchars_ = 0;
93  continue;
94  }
95 
96  if(no_spaces_left) {
97  (*m_os_) << sentence;
98  break;
99  }
100 
101  // See if we can fit more on the line
102  while(offset != std::string::npos) {
103  auto temp = sentence.find(' ', offset + 1);
104  if(temp > char_left || temp == std::string::npos) break;
105  offset = temp;
106  }
107 
108  (*m_os_) << sentence.substr(0, offset);
109  (*m_os_) << std::endl;
110  m_nchars_ = 0;
111  sentence = sentence.substr(offset + 1); // swallow the space
112  }
113  // Print a newline if we have more lines left
114  if(si < nsentences - 1) {
115  (*m_os_) << std::endl;
116  m_nchars_ = 0;
117  }
118 
119  } // loop over lines
120  return n;
121  }
122 
123  int overflow(std::streambuf::int_type c) override {
124  if(m_nchars_ == m_w_) {
125  (*m_os_) << std::endl;
126  m_nchars_ = 0;
127  }
128  (*m_os_) << std::streambuf::traits_type::to_char_type(c);
129  ++m_nchars_;
130 
131  return c;
132  }
